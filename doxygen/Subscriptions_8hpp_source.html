349 {
350  private:

352  std::string m_name;
353  std::string m_units;

355  std::function<void(int, Time)> update_callback;
356  std::vector<X> vals;
357  public:
358  VectorSubscription () noexcept {};
359 
366  VectorSubscription (bool required,
367  ValueFederate *valueFed,
368  std::string name,
369  int startIndex,
370  int count,
371  const X &defValue,
372  const std::string &units = "")
373  : fed (valueFed), m_name (std::move (name)), m_units (units)
374  {
375  ids.reserve (count);
376  vals.resize (count, defValue);
377  if (required)
378  {
379  for (auto ind = startIndex; ind < startIndex + count; ++ind)
380  {
381  auto id = fed->registerRequiredSubscription<X> (m_name, ind, m_units);
382  ids.push_back (id);
383  }
384  }
385  else
386  {
387  for (auto ind = startIndex; ind < startIndex + count; ++ind)
388  {
389  auto id = fed->registerOptionalSubscription<X> (m_name, ind, m_units);
390  ids.push_back (id);
391  }
392  }
393  fed->registerSubscriptionNotificationCallback (ids, [this](subscription_id_t id, Time tm) {
394  handleCallback (id, tm);
395  });
396  }

409  {
410  }

413  : fed (vs.fed), m_name (std::move (vs.m_name)), m_units (std::move (vs.m_units)), ids (std::move (vs.ids)),

415  {
416  // need to transfer the callback to the new object

418  handleCallback (id, tm);
419  });
420  };
422  VectorSubscription &operator= (VectorSubscription &&vs) noexcept

429  vals = std::move (vs.vals);
430  // need to transfer the callback to the new object
431  fed->registerSubscriptionNotificationCallback (ids, [this](subscription_id_t id, Time tm) {
432  handleCallback (id, tm);
433  });
434  return *this;
435  }
438  const std::vector<X> &getVals () const { return vals; }
442  const X &operator[] (int index) const { return vals[index]; }
448  void registerCallback (std::function<void(int, Time)> callback) { update_callback = std::move (callback); }
449 
450  private:
451  void handleCallback (subscription_id_t id, Time time)
452  {
453  X out;
454  auto res = std::lower_bound (ids.begin (), ids.end (), id);
455  int index = static_cast<int> (res - ids.begin ());
456  fed->getValue (ids[index], out);
457  vals[index] = out;
458  if (update_callback)
459  {
460  update_callback (index, time);
461  }
462  }
